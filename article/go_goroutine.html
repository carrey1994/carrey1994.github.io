Goroutine & Channel

 Goroutine 
func main() {
    go hello() //多開一個協程(Goroutine)處理hello()
    for {
        time.Sleep(time.Second)
        //將main thread睡眠以看goroutine執行hello()
        //否則goroutine還未完成，main thread及結束，程式則以先中止
    }
}
func hello() {
    fmt.Println("hello")
}
Channel 
func main() {
    c := make(chan int)
    go pass(c)
    x := <-c //main thread等待著channel來的消息
    fmt.Println(x)
}
func pass(c chan int) {
    time.Sleep(time.Second * 3) //因main thread等待著，之後3秒睡完
    fmt.Println("pass") //pass字串輸出
    c <- 100 //訊息送進channel
}
func passB(c chan int) {
    //若呼叫的為passB，則當c送出消息後，main thread可能接到後並結束程式
    //所以console可能未出現"pass"
    time.Sleep(time.Second * 3)
    c <- 100
    fmt.Println("pass")
}
