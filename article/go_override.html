Go Override應用

package main import "fmt" type IGame interface { Clone() IGame Cost() GetAmount() int } type Game struct { IGame Money int } func (g *Game) GetAmount() int { return g.Money } func (g *Game) Cost() { g.Money -= 100 } type PCGame struct { Game } func (g *PCGame) Cost() { //若無override則依原先Game一般Money-100 g.Money -= 200 } func main() { var game IGame = &PCGame{Game{Money: 1000}} game.Cost() fmt.Println(game.GetAmount()) }
